捕捉事件触发器(distributor$)推送的指定类型的事件，并返回一个observable

Example
import { attract, distributor$ } from "rx-samsara";
import { of } from "rxjs";
import { switchMap } from "rxjs/operators";

const obs$ = attract("testEvent").pipe(switchMap(event => {
  return of("testData");
}));

obs$.subscribe(val => {
  console.log(val);  // log: testData
});

distributor$.next("testEvent");

permeate(observablesMap, inputOptions)

Source:
  • permeate.js, line 32

observable与react组件的集成(将observable转换为组件属性)

Example
import { permeate } from "rx-samsara";
import { from } from "rxjs";

const testData$ = from(
  new Promise(resolve => {
    setTimeout(() => {
      resolve(123);
    }, 800);
  })
};

class CompA extends React.Component {
  render() {
    // this.props.testData === 123
    return <div>{this.props.testData}</div>;
  }
}

export default permeate({ testData: testData$ })(CompA);
